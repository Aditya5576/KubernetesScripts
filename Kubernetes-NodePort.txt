Kubernetes-NodePort

Service + Deployment : Kubernetes Networking

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: flipkart
spec:
  replicas: 1
  selector:
    matchLabels:
      name: deployment
  template:
    metadata:
      name: adi
      labels:
        name: deployment
    spec:
      containers:
       - name: signup
         image: httpd
         ports:
           - containerPort: 80
---
apiVersion: v1
kind: Service
metadata:
  name: flipkartsignup
spec:
  ports:
    - port: 80
      targetPort: 80
  selector:
    name: deployment
  type: NodePort
